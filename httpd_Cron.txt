import os
import commands
import sys

service_name = "httpd"
def check_service_status(service_name):

        status = os.system('service '+service_name+ ' status > /dev/null')
        return status
file = open('test1.txt','a')


def main():

        if (check_service_status(service_name) == 0):
                st = "HTTPD UP!! "
                file.write(st + '\n')
                file.close()
#print (st, file=f)
        else:
                st = "HTTPW DOWN!! "
                file.write(st + '\n')
                file.close()
main()
